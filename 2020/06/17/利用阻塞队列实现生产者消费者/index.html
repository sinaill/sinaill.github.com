利用阻塞队列实现生产者消费者

发表于 2020-06-17 | 分类于 java基础

实现方法

利用阻塞队列中的两个方法：

put(anObject):把anObject加到BlockingQueue里,如果BlockQueue没有空间,则调用此方法的线程被阻断
take():取走BlockingQueue里排在首位的对象,若BlockingQueue为空,阻断进入等待状态直到BlockingQueue有新的数据被加入:

public class PCtest {
	public static void main(String[] args) {
		BlockingQueue<Object> queue = new LinkedBlockingQueue<Object>(100);
		for(int i = 0;i < 10;i++){
			new producer("生产者"+i,queue).start();
		}
		for(int i = 0;i < 10;i++){
			new consumer("消费者"+i,queue).start();
		}
	}
}


class producer extends Thread{
	BlockingQueue<Object> queue;
	public producer() {
		super();
		// TODO Auto-generated constructor stub
	}

	public producer(String name,BlockingQueue<Object> queue) {
		super(name);
		// TODO Auto-generated constructor stub
		this.queue = queue;
	}

	@Override
	public void run() {
		// TODO Auto-generated method stub
		while(true){
			
			try {
				queue.put(new Object());
				System.out.println(getName()+"生产一件商品");
				producer.sleep(1000);
			} catch (Exception e) {
				// TODO Auto-generated catch block
				e.printStackTrace();
			}
		}
	}
	
}

class consumer extends Thread{
	BlockingQueue<Object> queue;
	public consumer() {
		super();
		// TODO Auto-generated constructor stub
	}

	public consumer(String name,BlockingQueue<Object> queue) {
		super(name);
		// TODO Auto-generated constructor stub
		this.queue = queue;
	}

	@Override
	public void run() {
		// TODO Auto-generated method stub
		while(true){
			
			try {
				queue.take();
				System.out.println(this.getName()+"消费1件商品");
				consumer.sleep(1000);
			} catch (InterruptedException e) {
				// TODO Auto-generated catch block
				e.printStackTrace();
			}
		}
		
	}
	
}

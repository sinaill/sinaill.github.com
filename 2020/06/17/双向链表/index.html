基本操作

头插法

public void insertNodeHead(Node<T> node){
	if(first == null){
		first = node;
		last = node;
		return;
	}
	first.setPre(node);//将第一个节点的pre设为新增进来的节点
	node.setNext(first);//将第一个节点设为新增节点的next
	first = node;//将第一个节点置换为新增进来的节点
}

尾插法

public void insertNodeTail(Node<T> node){
	
	Node<T> temp = first;
	
	if(first ==  null){
		first = node;
		last = node;
		return;
	}
	
	while(temp.getNext() != null){
		temp = temp.getNext();
	}
	temp.setNext(node);	
	node.setPre(temp);
	last = node;
}

public T getData(int index){
	if(isEmpty()){
		return null;

	Node<T> temp = first;
	while(temp != null && --index > 0){
		temp = temp.getNext();
	}

public void removeNodeByPos(int index){
	if(isEmpty()){
		return;
	}
	if(getLength()==index){
		removeLastNode();
		return;
	}
	Node<T> temp = first;
	while(temp != null && --index > 0){
		temp = temp.getNext();
	}
	temp.getNext().setPre(temp.getPre());
	temp.getPre().setNext(temp.getNext());
}

指定位置插入节点

public void insertByPos(int index,Node<T> node){
	Node<T> temp = first;
	if(isEmpty() || index == 1){//空链表或者插入第一个时用头插法
		insertNodeHead(node);
		return;
	}
	if(getLength()+1 == index){//指定最后位置时用尾插法
		insertNodeTail(node);
		return;
	}
	while(temp != null && --index > 0){
		temp = temp.getNext();
	}
	node.setNext(temp);//新节点的next指向当前节点
	node.setPre(temp.getPre());//新节点的pre指向上一个节点
	temp.getPre().setNext(node);//上一个节点的next指向新节点
}

倒置空链表

public void reverse(){
	if(isEmpty()){
		return;
	}
	Node<T> newNode = first;
	Node<T> temp = first;
	setFirst(null);
	setLast(null);
	while(newNode != null){//遍历双链表
		temp = newNode.getNext();
		newNode.setPre(null);
		newNode.setNext(null);//置空前后节点作为新节点
		insertNodeHead(newNode);//头插法
		newNode = temp;
	}
}

查看双链表长度

public int getLength(){
	Node<T> temp = first;
	int length = 0;
	while(temp != null){
		length++;
		temp = temp.getNext();
	}
	return length;
}

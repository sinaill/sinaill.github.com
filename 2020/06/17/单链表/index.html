单链表的概念

单链表每个节点node中存储了下一个节点的引用

结构

节点

class Node<T>{
	private T data;
	private Node<T> next;//下一个节点的引用
	
	public Node() {
	}
	
	public Node(T data) {
		super();
		this.data = data;
	}

	public T getData() {
		return data;
	}
	public void setData(T data) {
		this.data = data;
	}
	public Node<T> getNext() {
		return next;
	}
	public void setNext(Node<T> next) {
		this.next = next;
	}
	
	
}

	单链表基本操作

单链表的基本操作

头插法插入节点

public void insertNodeHead(Node<T> node){
	node.setNext(header);
	header = node;
}

尾插法插入节点

public void insertNodeTail(Node<T> node){
	Node<T> temp = header;
	if (header == null) {
		header = node;
		return;
	}
	while(temp.getNext() != null){
		temp = temp.getNext();
	}
	temp.setNext(node);
}

移除第一个节点

public void removeFirstNode(){
	if(isEmpty()){
		return;
	}
	header = header.getNext();
}

移除最后一个节点

public void removeLastNode(){
	if(isEmpty()){
		return;
	}
	Node<T> temp = header;
	while(temp.getNext().getNext() != null){
		temp = temp.getNext();
	}
	temp.setNext(null);
}

获取单链表长度

public int getLength(){
	int tempLength = 0;
	Node<T> tempNode = header;
	while(tempNode != null){
		tempLength++;
		tempNode = tempNode.getNext();
	}
	return tempLength;
}

显示所有节点信息

public void dislayAllNode(){
	if(isEmpty()){
		return;
	}
	Node<T> temp = header;
	while(temp != null){
		System.out.println(temp.getData());
		temp = temp.getNext();
	}
}

指定位置插入节点

public void insertNodeByPos(int pos,Node<T> node){
	//链表为空或者指定位置超过链表长度结束
	if(isEmpty() || getLength()+1 < pos){
		return;
	}
	if(pos == 1){
		insertNodeHead(node);
	}else if (pos == getLength()+1) {
		insertNodeTail(node);
	}else {
		Node<T> temp = header;
		for (int i = pos - 2; i > 0 && temp != null; i--) {
			temp = temp.getNext();
		}
		node.setNext(temp.getNext());
		temp.setNext(node);
	}
}

删除指定位置节点

public void deleteNodeByPos(int pos){
	if(isEmpty()){
		return;
	}
	Node<T> temp = header;
	if (pos == 1) {
		removeFirstNode();
		return;
	}

	for (int i = pos - 2; i > 0 && temp != null; i--) {
		temp = temp.getNext();
	}
	temp.setNext(temp.getNext().getNext());
}

查看元素在链表中位置

public Integer elementPos(T data){
	if(isEmpty()){
		return null;
	}
	Node<T> temp = header;
	int pos = 1;
	while(temp != null){
		if(temp.getData().equals(data)){
			return Integer.valueOf(pos);
		}
		pos++;
		temp = temp.getNext();
	}
	return null;
}

倒置链表

public void reverse(){
	if(isEmpty() && getLength()==1){
		return;
	}	
	Node<T> next;//保存下一个节点
	Node<T> temp = header;//当前处理节点
	header = null;//第一次做特殊处理，使尾节点中的next为空
	while(temp != null){
		next = temp.getNext();
		temp.setNext(header);
		header = temp;
		temp = next;
	}
}

查看链表是否为空

public boolean isEmpty(){
	return Objects.isNull(header);
}

清空链表

public void clear(){
	header = null;
}
